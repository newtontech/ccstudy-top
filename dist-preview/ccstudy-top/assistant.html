Claude Code зҡ„жҢҒд№…еҢ– AI еҠ©жүӢжЁЎејҸ вҖ” жҢҒз»ӯиҝҗиЎҢгҖҒдё»еҠЁиЎҢеҠЁгҖҒзІҫз®ҖдәӨдә’

KAIROS жҰӮиҝ°

Always-on зҡ„жҢҒд№…еҢ– AI еҠ©жүӢ

KAIROS жҳҜ Claude Code зҡ„жҢҒд№…еҢ– AI еҠ©жүӢжЁЎејҸгҖӮдёҺдј з»ҹзҡ„"иҜ·жұӮ-е“Қеә”"ејҸдәӨдә’дёҚеҗҢпјҢ KAIROS д»Ҙ Always-on ж–№ејҸжҢҒз»ӯиҝҗиЎҢпјҢзӣ‘жҺ§зҺҜеўғеҸҳеҢ–е№¶дё»еҠЁйҮҮеҸ–иЎҢеҠЁгҖӮе®ғе°ұеғҸдёҖдёӘе§Ӣз»ҲеңЁзәҝзҡ„ејҖеҸ‘дјҷдјҙпјҢ еңЁдҪ зј–еҶҷд»Јз Ғзҡ„еҗҢж—¶й»ҳй»ҳе®ҲжҠӨйЎ№зӣ®еҒҘеә·гҖӮ

KAIROS зҡ„ж ёеҝғи®ҫи®ЎзҗҶеҝөеҢ…еҗ«дёүдёӘе…ій”®зү№жҖ§пјҡжҢҒд№…еҢ–дјҡиҜқдҝқжҢҒи·Ёз»Ҳз«Ҝзҡ„зҠ¶жҖҒиҝһз»ӯжҖ§пјӣ зҺҜеўғзӣ‘жҺ§иғҪеҠӣдҪҝе…¶иғҪдё»еҠЁеҸ‘зҺ°й—®йўҳпјӣиҖҢ з®ҖиҰҒжЁЎејҸпјҲBrief Modeпјү еҲҷйҖҡиҝҮи¶…з®ҖжҙҒе“Қеә”еҮҸе°‘и®ӨзҹҘиҙҹжӢ…пјҢи®©жҢҒд№…дёҠдёӢж–Үдёӯзҡ„дәӨдә’жӣҙеҠ й«ҳж•ҲгҖӮ

KAIROS еҠ©жүӢжһ¶жһ„

дјҡиҜқз®ЎзҗҶ

и·Ёз»Ҳз«ҜдјҡиҜқзҡ„зҠ¶жҖҒжҢҒд№…еҢ–

KAIROS зҡ„дјҡиҜқз®ЎзҗҶжҳҜе…¶жҢҒд№…еҢ–иғҪеҠӣзҡ„ж ёеҝғгҖӮйҖҡиҝҮ жҢҒд№…еҢ–дјҡиҜқ жңәеҲ¶пјҢKAIROS иғҪеӨҹи·Ёи¶Ҡз»Ҳз«ҜдјҡиҜқдҝқжҢҒе®Ңж•ҙзҡ„зҠ¶жҖҒдҝЎжҒҜгҖӮеҚідҪҝдҪ е…ій—ӯдәҶз»Ҳз«ҜзӘ—еҸЈпјҢ дёӢж¬ЎеҗҜеҠЁж—¶ KAIROS еҸҜд»ҘиҮӘеҠЁжҒўеӨҚд№ӢеүҚзҡ„дёҠдёӢж–ҮпјҢз»§з»ӯжңӘе®ҢжҲҗзҡ„е·ҘдҪңгҖӮ

дјҡиҜқеҺҶеҸІи®°еҪ•дәҶе®Ңж•ҙзҡ„дәӨдә’иҝҮзЁӢпјҢеҢ…жӢ¬еҜ№иҜқж¶ҲжҒҜгҖҒж–Үд»¶ж“ҚдҪңгҖҒе‘Ҫд»Өжү§иЎҢзӯүпјҢ дҪҝеҫ— KAIROS иғҪеӨҹеңЁд»»ж„Ҹж—¶й—ҙзӮ№еҮҶзЎ®зҗҶи§ЈйЎ№зӣ®зҡ„еҪ“еүҚзҠ¶жҖҒе’ҢеҺҶеҸІеҸҳжӣҙгҖӮ

жӯҘйӘӨ 1/3

еҲӣе»әж–°зҡ„ KAIROS дјҡиҜқгҖӮжҜҸдёӘдјҡиҜқжӢҘжңүе”ҜдёҖ IDпјҢи®°еҪ•еҗҜеҠЁж—¶й—ҙе’ҢеҲқе§ӢдёҠдёӢж–ҮпјҢдёәеҗҺз»ӯжҢҒд№…еҢ–жҒўеӨҚеҘ е®ҡеҹәзЎҖгҖӮ

дё»еҠЁиЎҢеҠЁ

зҺҜеўғзӣ‘жҺ§дёҺжҷәиғҪе“Қеә”

KAIROS жңҖејәеӨ§зҡ„зү№жҖ§д№ӢдёҖжҳҜе…¶дё»еҠЁиЎҢеҠЁиғҪеҠӣгҖӮе®ғжҢҒз»ӯзӣ‘жҺ§ејҖеҸ‘зҺҜеўғзҡ„еҗ„з§ҚеҸҳеҢ–пјҢ еҢ…жӢ¬ж–Үд»¶дҝ®ж”№гҖҒGit зҠ¶жҖҒеҸҳжӣҙгҖҒжөӢиҜ•з»“жһңзӯүпјҢе№¶еңЁжЈҖжөӢеҲ°жҪңеңЁй—®йўҳж—¶иҮӘеҠЁжҸҗдҫӣе»әи®®жҲ–жү§иЎҢдҝ®еӨҚгҖӮ

зҺҜеўғзӣ‘жҺ§

жҢҒз»ӯиҝҪиёӘж–Үд»¶еҸҳеҢ–гҖҒGit зҠ¶жҖҒгҖҒжөӢиҜ•з»“жһңзӯүејҖеҸ‘зҺҜеўғжҢҮж Ү

йҖҡиҝҮж–Үд»¶зі»з»ҹзӣ‘еҗ¬е’Ң Git й’©еӯҗпјҢKAIROS е®һж—¶ж„ҹзҹҘйЎ№зӣ®дёӯзҡ„жҜҸдёҖж¬ЎеҸҳжӣҙпјҢжһ„е»әе®Ңж•ҙзҡ„еҸҳжӣҙдёҠдёӢж–ҮгҖӮ

дё»еҠЁе»әи®®

жЈҖжөӢеҲ°й—®йўҳж—¶иҮӘеҠЁеҲҶжһҗе№¶жҸҗдҫӣе»әи®®ж–№жЎҲ

еҲ©з”Ёд»Јз ҒеҲҶжһҗе’ҢжЁЎејҸеҢ№й…ҚпјҢKAIROS еңЁй—®йўҳжү©еӨ§еүҚе°ұиғҪеҸ‘зҺ°жҪңеңЁйЈҺйҷ©пјҢе№¶дё»еҠЁеҗ‘ејҖеҸ‘иҖ…жҠҘе‘ҠгҖӮ

иҮӘеҠЁдҝ®еӨҚ

еңЁз”ЁжҲ·жҺҲжқғзҡ„жғ…еҶөдёӢиҮӘеҠЁжү§иЎҢдҝ®еӨҚж“ҚдҪң

еҜ№дәҺеёёи§Ғй—®йўҳпјҲеҰӮ lint й”ҷиҜҜгҖҒж јејҸй—®йўҳпјүпјҢKAIROS еҸҜд»ҘеңЁиҺ·еҫ—жҺҲжқғеҗҺиҮӘеҠЁдҝ®еӨҚпјҢеҮҸе°‘жүӢеҠЁе№Ійў„гҖӮ

kairos/proactive.ts
// дё»еҠЁиЎҢеҠЁеј•ж“Һ
async function monitorEnvironment() {
  const changes = await detectFileChanges();
  const issues = await analyzeChanges(changes);

  if (issues.length > 0) {
    // дё»еҠЁеҗ‘з”ЁжҲ·жҠҘе‘ҠеҸ‘зҺ°зҡ„й—®йўҳ
    notifyUser({
      type: 'proactive_insight',
      message: `еҸ‘зҺ° ${issues.length} дёӘжҪңеңЁй—®йўҳ`,
      suggestions: issues.map(generateFix),
    });
  }
}

и¶…з®ҖжҙҒе“Қеә”пјҢжҸҗеҚҮжҢҒз»ӯдәӨдә’ж•ҲзҺҮ

з®ҖиҰҒжЁЎејҸжҳҜ KAIROS й’ҲеҜ№жҢҒд№…еҢ–дёҠдёӢж–ҮеңәжҷҜдё“й—ЁдјҳеҢ–зҡ„е“Қеә”йЈҺж јгҖӮеңЁй•ҝж—¶й—ҙиҝҗиЎҢдёӯпјҢ иҝҮеӨҡзҡ„иҫ“еҮәдјҡйҖ жҲҗдҝЎжҒҜиҝҮиҪҪгҖӮз®ҖиҰҒжЁЎејҸйҖҡиҝҮ зІҫз®Җиҫ“еҮә еӨ§е№…еҮҸе°‘е“Қеә”й•ҝеәҰпјҢеҸӘдҝқз•ҷжңҖе…ій”®зҡ„дҝЎжҒҜпјҢжҳҫи‘—жҸҗй«ҳжҢҒз»ӯдәӨдә’зҡ„ж•ҲзҺҮгҖӮ

жҷ®йҖҡжЁЎејҸ

жҲ‘е·Із»Ҹд»”з»ҶжЈҖжҹҘдәҶдҪ зҡ„д»Јз ҒпјҢеҸ‘зҺ° src/app.ts ж–Үд»¶з¬¬42иЎҢжңүдёҖдёӘжңӘеӨ„зҗҶзҡ„й”ҷиҜҜгҖӮ е»әи®®ж·»еҠ  try-catch еқ—жқҘжҚ•иҺ·еҸҜиғҪзҡ„ејӮеёёгҖӮиҝҷйҮҢжҳҜдҝ®еӨҚд»Јз Ғ...

з®ҖиҰҒжЁЎејҸ

src/app.ts:42 - зјәе°‘ try-catchпјҢе»әи®®ж·»еҠ й”ҷиҜҜеӨ„зҗҶ

з®ҖиҰҒжЁЎејҸиҫ“еҮәжӣҙзІҫзӮјпјҢйҖӮеҗҲжҢҒд№…дёҠдёӢж–Үдёӯеҝ«йҖҹиҺ·еҸ–е…ій”®дҝЎжҒҜ
kairos/briefMode.ts
// з®ҖиҰҒжЁЎејҸй…ҚзҪ®
interface BriefModeConfig {
  enabled: boolean;
  maxLength: number;       // е“Қеә”жңҖеӨ§еӯ—з¬Ұж•°
  showCodeContext: boolean; // жҳҜеҗҰжҳҫзӨәд»Јз ҒдёҠдёӢж–Ү
  verbosity: 'minimal' | 'compact' | 'normal';
}

// з®ҖиҰҒжЁЎејҸдёӢзҡ„иҫ“еҮәж јејҸеҢ–
function briefFormat(issue: CodeIssue): string {
  return `${issue.file}:${issue.line} - ${issue.summary}`;
}

// жҷ®йҖҡжЁЎејҸдёӢзҡ„иҫ“еҮәж јејҸеҢ–
function normalFormat(issue: CodeIssue): string {
  return `жҲ‘е·Із»Ҹд»”з»ҶжЈҖжҹҘдәҶдҪ зҡ„д»Јз ҒпјҢеҸ‘зҺ° ${issue.file} ж–Үд»¶`
    + `з¬¬${issue.line}иЎҢжңүдёҖдёӘжңӘеӨ„зҗҶзҡ„й”ҷиҜҜгҖӮ`
    + `е»әи®®ж·»еҠ  try-catch еқ—жқҘжҚ•иҺ·еҸҜиғҪзҡ„ејӮеёё...`;
}
